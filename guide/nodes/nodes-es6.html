# ES6

# var的弊端及let关键字

# var的弊端

  • var声明的变量有预解析，容易造成逻辑混乱，可以先使用，后声明。
  • var可以重复定义同一个变量，容易造成逻辑错误，第二次应该是修改变量，而不是定义
  • var在for循环中，造成for循环的污染问题
  • var声明的变量没有块级作用域（ES5中的作用域：全局和局部）

# let关键字

let的出现就是为了解决var的弊端

  • 必须先声明再使用
  • 不能重复声明同一变量
  • 不存在for循环污染问题
    • 用let定义的i只能在for循环的内部去使用，for循环结束，变量被回收掉。
  • 存在块级作用域

# const声明常量

  • 一旦声明，必须赋值
  • 一旦赋值，（值类型）不能被修改；（引用类型）其值可变，因为引用类型绑定的是内存地址
  • 常量名一般都大写
  • const也用来模块化中引入模块的一个关键字，作为模块导入可以用小写
    • const express = require('express')

# 解构语法

# 对象的解构

let obj = {
  name:'node',
  age:11,
  job:"web"
}
//如何获取对象中的每一个值
//方法1
let name = obj.name;
let age = obj.age;
let job = obj.job;
console.log(name,age,job);
//方法2
let name = obj['name'];
let age = obj['age'];
let job = obj['job'];
console.log(name,age,job);
//方法3 => 解构对象
let {name,age,job} = obj;
console.log(name,age,job);

//为避免变量名被声明过，起冲突。可以对大括号内的变量起别名
let {name:myname} = obj;
console.log(myname);


//注意点
/*
	1.let {name,age,job} = obj
		相当于
			let name = obj.name;
			let age = obj.age;
			let job = obj.job;
	2.解构的大括号内部的属性名可以调换顺序
		let {age,name,job} = obj
		打印的结果与属性名是一一对应的
	3.可以对对象进行部分解构
		let {name} = obj
	4.大括号里面的属性名必须是对象的属性名，不然会报undefined
	5.可以给定义出来的变量换一个名字
		let {name:myname} = obj
		相当于
			let myname = obj.name
*/

# 函数参数的解构赋值

什么时候用函数参数的解构赋值呢？

当函数调用中的实参为对象或数组时

let obj = {
  name:'node',
  age:11,
  job:"web"
}

//1.参数的解构赋值
function fn({name,age,job}){  //{}看作一个整体，属于一个形参;{name,age,job} = obj  参数的解构赋值
  console.log(name);
  console.log(age);
  console.log(job);
}

fn(obj)	//node 11 web  【传入一个对象，用解构的方式获取它的属性】
fn()	//Cannot destructure property 'name' of 'undefined' as it is undefined. 无法解构未定义的属性name，因为它是undefined

//2.参数的整体默认值
function fn({name,age,job}={}){ 
  console.log(name);
  console.log(age);
  console.log(job);
}
fn()	//undefined undefined undefined

//3.需要解构的参数指定默认值
function fn({name='su',age=18,job='web'}={}){ //不为整体赋默认值报错
  console.log(name);
  console.log(age);
  console.log(job);
}
fn()	//su 18 web

/*
	1.定义函数的时候，只写了fn({name,age,job})，调用的时候fn(obj)，相当于{name,age,job} = obj
	2.定义函数的时候，只写了fn({name,age,job})，调用的时候fn()，相当于{name,age,job} = null
	3.为了防止调用的时候没有传参报错，可以写成fn({name,age,job}={})
*/

# 三点运算符-剩余参数

...rest 是ES6提供的新方法，收集剩余的实参，并将其放入一个数组中，rest就是这个数组，是一个真正的数组，可以使用数组的方法。【rest是个变量，其名可换，重点是...不能多也不能少】

注意：rest参数只能放到最后，否则会报错。

与arguments的区别：arguments是伪数组，用于获取全部参数，箭头函数没有arguments

//arguments的写法
function fn() {
  console.log(arguments);
  //获取每一个值
  console.log(arguments[0]);
  console.log(arguments[1]);
  console.log(arguments[2]);
  //遍历
  for (let i = 0; i < arguments.length; i++) {
    console.log(arguments[i]);
  }
}
fn(10, 20, 30)	//[Arguments] { '0': 10, '1': 20, '2': 30 }

//普通方法
function fn(a,b,c){
  console.log(a);
  console.log(b);
  console.log(c);
}
fn(10,20,30)	//10 20 30 

//..rest方法
function fn(...rest){
  console.log(rest);//使用rest参数搭配的变量是一个数组，该变量将多余的参数放入数组中
  //获取每一个参数
  console.log(rest[0]);
  console.log(rest[1]);
  console.log(rest[2]);
  //foreach遍历
  rest.forEach((item,index)=>{
    console.log(item,index)
  })
}
fn(10,20,30)

# 三点运算符-扩展运算符

它的作用就是可以将数组或者对象展开，拆开成为一个一个单独的数据。

let arr = [10,20,30]
console.log(...arr) //引申出：数组中的值作为函数调用的实参使用


/*
	使用场景
		1.数组中的值作为函数调用的实参使用
		2.合并数组
		3.合并对象
		4.es6中另一个合并对象的方法
			Object.assign({}, obj, obj2)
*/

//1.数组中的值作为函数调用的实参使用
let str1 = [10,20,30]
function fn(a,b,c){
  console.log(a,b,c)
}
fn(str1)	//[10,20,30] undefined undefined
fn(str1[0],str1[1],str1[2])
fn(...str1)	//把数组扩展开来，相当于直接去掉[]

//终极简写
let str1 = [10,20,30]
function fn(...rest){
  console.log(rest[0],rest[1],rest[3])
}
fn(...str1)

//因为它可以将数组和对象扩展成一个个单独的数据，我们可以用来做数组和对象的合并
//2.合并数组
let str2 = [1,2,3]
let str3 = [4,5,6]
let newStr = [...str2,...str3]
console.log(newStr)	//[1,2,3,4,5,6]

//3.合并对象
let obj1 = {
  name:'su',
  age:18
}
let obj2 = {
  gender:'男',
  school:'沈阳师范大学'
}
let newObj = {...obj1,...obj2}
console.log(newObj)

//4.ES6提供的对象合并方法
let newObj1 = Object.assign({}, obj1, obj2);
console.log(newObj1)

# 箭头函数

ES6允许使用箭头函数（ => ）简化函数的定义。

箭头函数本质也是函数，它出现的目的是为了让我们在使用回调函数的时候更简单。

//之前的写法
function fn(){
  console.log(123)
}

//箭头函数的写法
let fn=()=>{
  console.log(123)
}


/*
	注意
		1.函数体只有一句代码时，可以省略大括号不写
			let fn = () => console.log(123)
		2.函数体只有一句代码时，且是返回一个结果时，可以省略return
			let fn = (x) => x+100
		3.当形参只有一个的时候，小括号可以省略
			let fn = x => x+100
		4.特殊点：当函数返回值是一个对象的时候，不要省略大括号和return
			let fn = () => {
				return {name:'su',age:18}
			}
		5.箭头函数不可以使用arguments获取参数列表，可以使用rest参数代替。
		6.箭头函数中this的指向，指向外层作用域
*/

//普通函数
function People(name,age){
  this.name=name;
  this.age=age;
  this.say=function(){
    let _this=this;
    setTimeout(function(){
  		console.log(this.name);	//this指向window，所以为undefined
      console.log(_this.name);	//张三
		})
  }
}

let p = new People('张三',11);
p.say()

//箭头函数
function People(name,age){
  this.name=name;
  this.age=age;
  this.say=function(){
    setTimeout(()=>{
  		console.log(this.name);	//张三
		})
  }
}

let p = new People('张三',11);
p.say()

# Promise

Promise是异步编程的一种解决方案

它可以避免回调地狱（异步回调的多层嵌套），把异步代码改成调用起来像同步一样。

一个Promise对象有以下几种状态：

  • pending 初始状态

  • fulfilled 操作成功

  • rejected 操作失败

Promise有两个特点

  1. 对象的状态不受外界影响（Promise对象代表一个异步操作，只有异步操作的结果可以决定当前的状态，除此之外，任何其他操作都无法改变这个状态）

  2. 一旦状态改变，就不会再变，任何时候都可以得到这个结果。（Promise状态改变只有两种可能：从初始到成功和从初识到失败）

// 默认pending: 初始状态
var p=new Promise(function (resolve,reject) {
    if("操作成功"){
        resolve();//pending-->fulfilled  异步操作成功的回调函数
    }else{
        reject(); //pending-->reject     异步操作失败的回调函数
    }
})
p.then(data => {//在外面调用then处理成功的逻辑
    console.log("处理成功的逻辑");//fulfilled  
}).catch(err=>{//在外面调用catch处理失败的逻辑
    console.log("失败的逻辑");//reject
})
// then方法会在异步成功后调用，catch方法会在异步失败后调用
