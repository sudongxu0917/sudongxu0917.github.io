# 公共配置项

// 公共配置项
var option = {
  // 标题
  title: {
    text: "标题",
    link:"https://www.xiaosutongxue.com"
    textStyle: {
      color: "#72f4f1",
    },
  },
  // 提示框
  tooltip: {
    trigger:"axis",	//设置触发提示的项目，默认放在内容上，可以改成坐标轴上
    textStyle: {
      color: "#fff",
    },
    // 设置整个框的样式，如果不这么设置，框是带黑边框的
    extraCssText:"width:200px,height:100px,background:transparent,border:none"
    formatter:(v)=>{
      console.log(v)
      return xxxx;
    }
  },
  // 工具箱
  toolbox:{
    feature:{
      saveAsImage:{name:'',background:''},	// 另存为图片
      dataView:{},	// 数据视图（临时修改内容）
      restore:{},	// 重置
      dataZoom:{},	// 区域缩放
      magicType:{	// 魔术类型 几个图之间的数据格式是一样的，一般在柱状图和折线图间来回切换
        type:['line','bar']	
      }
    }
  },  
  // 图例组件
  legend: {
    data: ["销量"],
    itemHeight: 16, // 图例的高度
    itemWidth: 16, // 图例的宽度
    inactiveColor: "rgba(0,0,0,0.3)", // 图例关闭后的颜色
    textStyle: {
    	fontSize: 14,
      color: "#fff", // 图例的文字颜色
      padding: [0, 0, 0, 0],
    },
    bottom: "0",
    left: "center",
  },
  // x轴
  xAxis: {
    data: ["衬衫", "羊毛衫", "雪纺衫", "裤子", "高跟鞋", "袜子"],
    //起点津贴边缘
    boundaryGap: false,
    //x轴文字的配置
    axisLabel: {
      show: true,
      interval: 0,
      textStyle: {
        color: "#fff",
      },
    },
    //X轴线的颜色以及宽度
    axisLine: {
      show: true,
      lineStyle: {
        color: "#ccc",
        type: "solid",
      },
    },
    // 分割线
    splitLine: {
      show: true,
      lineStyle: {
        type: "dashed",
        color: "rgba(255, 255, 255,0.3)",
      },
    },
  },
  // y轴
  yAxis: {
    scale: true,
  },
  // 系列列表：决定当前图形是什么图
  series: [],
};

// name:"省份名" value:"数值"
// 注意：name对应省份名一定要跟label显示出来的文字一一对应的
let d = [
  {
    name:"辽宁"
    value:"0"
  }
]
let option = {
  visualMap:{
    min:0,
    max:200,	// 只要超过200了，就算最深的颜色
    inRange:{
      color:["#fff","lightblue","darked"]	//可以设置两个或多个
    },
    calculable:true

  series:[
    name:'地图',
    type:"map"，
    mapType:"china",

    label:{
    	show:true
    },
    data:d,	// 指定上面的数据
  ]
}

option = {
  xAxis:{
  	type:"value",
  	scale:true  // 起点不为0
	},

    type:"value",
    scale:true  // 起点不为0

  series: [
    {
      name: "散点图",
      type: "scatter",	// scatter散点图 effectScatter 带涟漪效果的散点图
      rippleEffect:{	// 调整涟漪大小
        scale:20
      },
      showEffectOn:"emphasis",	// 鼠标悬浮后触发涟漪效果

      },

        color:()=>{}
      }

  ]
}
