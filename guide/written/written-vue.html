# 【2】路由守卫

// 路由前守卫
router.beforeEach((to, from, next) => {
  if (to.path == '/cart') {
    let token = localStorage.getItem('token')
    if (token) {
      next()
    } else {
      setTimeout(() => {
        next('/user')
      }, 1000)
    }
    return;
  }
  next()
})

# 【3】axios请求封装

request.js

//引入axios
import axios from 'axios'
//创建实例
const ins = axios.create({
    // 基础URL
    // baseURL: '/api',
    baseURL: 'http://api.xiaosutongxue.com:3000',
    //请求超时时间，单位是毫秒
    timeout: 5000
})

//请求拦截器：发送请求之前执行这个函数
ins.interceptors.request.use(config => {
    //config：本次请求的一些信息，如请求地址，请求超时时间，请求头等等
    let token = localStorage.getItem('token')
    if (token) {
        config.headers["X-Nideshop-Token"] = token
    }
    return config
}, err => {
    return Promise.reject(err)
})

//响应拦截器：在接收到响应之前，在执行then方法之前
//执行顺序 config => res => then
ins.interceptors.response.use(res => {
    // res：本次服务器响应的一些信息，返回的这个res被then的res形参接收了
    return res.data
}, err => {
    return Promise.reject(err)
})

//开放实例对象ins
export default ins
